##### INSTALLATION GUIDE #####
- Debian has a hercules (emulates IBM mainframe) install
- c3270: emulates IBM display stations

cmd:
sudo apt-get install hercules unrar c3270

create dir:
/mvs/downloads

cmd:
wget https://www.jaymoseley.com/hercules/downloads/archives/vs2StarterTapes.tar.gz
wget https://www.jaymoseley.com/hercules/downloads/archives/vs2DistributionTapes.tar.gz
wget https://www.jaymoseley.com/hercules/downloads/archives/mvsInstallationResources.tar.gz

cmd:
tar tzf downloads/mvsInstallationResources.tar.gz

extract cmd:
tar xzf downloads/mvsInstallationResources.tar.gz

cmd:
cd /mvs/tape
tar xzf ../downloads/vs2DistributionTapes.tar.gz

./set_console_mode
cd ..
./mvs

c3270 localhost:3270
press "F3" to enter the TSO prompt
type in "tsoappls" to return the to menu
type in "logoff" to log off the mainframe
type in "shutdown" to turn off the system
